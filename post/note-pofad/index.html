1 The smallest free number

accumArray :: Ix i => (e -> v -> e) -> e -> (i,i) -> [(i,v)] -> Array i e

$\lbrack (i,v)\rbrack$ 是 associating list, v 是 value, 不同于 entry.
e 是 initial entry. (i,i) 是 index 的上下界. 对上下界之间的所有 index 初始化 entry.
不是所有 index 都在 associating list $\lbrack(i,v)\rbrack$ 中出现过.
有的 index 从 associating list 中得到了 value, 这个 value 不是在 array 中 index 对应的 entry.
index 的 value 的作用是, 在 (e -> v -> e) 中改变 initial entry.
没有 value 以用于改变的 index 的 entry 就是 initial entry.

checklist :: [Int] -> Array Int Bool
checklist xs = accumArray (||) False (0,n) $ zip (filter (<= n) xs) (repeat True)
				where n = length xs
countlist :: [Int] -> Array Int Int
countlist xs = accumArray (+) 0 (0,n) $ zip xs $ repeat 1
sort xs = concat [replicate k x | (x,k) <- countlist xs]

divide and conquer:
把 xs 长度的一半, 作为 x 的值的二分点.
用 partition 将 xs 二分为 us 和 vs. 对每个 x 做一次 True/False 判定, 而不用排序.
若自身值小于长度一半的 x 的数量 (length us) 恰等于长度一半，说明 $\lbrack$ 0 .. length xs `div` 2 $\rbrack$ 之间的自然数是被取满了的,
则只需在 vs 中继续查找.

minfree xs			=	minfrom 0 (length xs , xs)
minfrom a (n,xs)	|	n == 0 		= a
					|	m == b - a 	= minfrom b (n-m , vs) -- 前半取满, 查找后半
					|	otherwise	= minfrom a (m , us)
					where (us , vs) = partition (< b) xs
							b 		= a + 1 + n `div` 2
							m 		= length us

2 A surpassing problem

这篇看得我像长这么大第一次见魔法。。

利用性质|法则进行形式推导

寻找 join
s.t. table (xs ++ ys) = join (table xs) (table ys) 形式成立.

-- table (xs ++ ys) = join txs tys 
--					= [(z,c + tcount z tys) | (z,c) <- txs] ++ tys

table (xs ++ ys) 
= [(z,scount z zs) | z : zs <- tails (xs ++ ys)]
-- tails (x:xs) = (x:xs) : tails xs
-- table xs = [(z,scount z zs) | z:zs <- tails xs]
= [(z,scountz zs) | z:zs <- map (flip . (++) ys) (tails xs) ++ tails ys]
-- by divide and conquer property
-- tails (xs ++ ys) = map (flip . (++) ys) (tails xs) ++ tails ys
= [(z,scount z (zs ++ ys)) | z:zs <- tails xs] ++ 
		[(z,scount z zs) | z:zs <- tails ys]
-- distributing <- over ++ 
-- z <- xs ++ ys = (z <- xs) ++ (z <- ys)
= [(z,scount z zs + scount z ys) | z:zs <- tails xs] ++ 
		[(z,scount z zs) | z:zs <- tails ys]
-- scount z (zs ++ ys) = scount z zs + scount z ys 
= [(z,c + scount z (map fst (table ys))) | (z,c) <- table xs] ++ table ys
= [(z,c + tcount z tys) | (z,c) <- txs] ++ tys
-- tys = table ys
-- tcount z tys = scount z (map fst tys)
= join txs tys

改进 tcount:

-- tcount z tys = length $ dropWhile ((<= z) . fst) tys

tcount z tys 
= scount z (map fst tys)
= length $ filter (> z) (map fst tys)
-- filter p $ map f = map f $ filter (p . f)
-- example: 
-- filter (<5) $ map (+1) [2,3,7,8]= map (+1) $ filter ((<5).(+1)) [2,3,7,8]
= length $ map fst $ filter ((> z) . fst) tys
-- length $ map f = length 
= length $ filter ((> z) . fst) tys
-- 改进定义, table已排序, dropWhile取代filter
-- table xs = sort [(z,scount z zs) | z:zs <- tails xs]
-- 相应地, join txs tys = [(x,c + tcount x tys) | (x,c) <- txs] `mergeSorted` tys
= length $ dropWhile ((<= z) . fst) tys

改进 join:

join txs  tys 
 = [(x , c + tcount x tys) | (x , c) <- txs] `mergeSorted` tys
-- 递归定义
join txs@((x , c) : txs') tys@((y , d) : tys') 
 = ((x , c + tcount x tys) : [( x , c + tcount x tys) | (x , c) <- txs']) `mergeSorted` tys
-- 比较 x, y, 决定哪个在先. 在递归地 comgpare - merge 中实现排序.  
-- 当 x < y, tcount x tys = length tys
 = (x , c + length tys) : join txs' tys
-- 当 x >= y, d = tcount x tys', tcount x tys = tcount x tys'
 = (y , d) : join txs tys'

最终:

table [x]   = [(x , 0)]
table xs    = join (m - n) (table ys) (table zs)
               where m = length xs
               n = m `div` 2
               (ys , zs) = splitAt n xs
join 0 txs [] = txs
join n [] tys = tys
join n txs@((x , c) : txs') tys@((y , d) : tys') 
			| x < y  = (x , c + n) : join n txs' tys
			| x >= y = (y , d) : join (n - 1) txs tys'

3 Improving on saddleback search

$invert \ f \ z$ is a list of all pairs $(x,y)$ satisfying $f(x,y) = z$.
$f$ is strictly increasing in each argument.

-- ver. Anne

-- invert f z = 
-- find (0 , z) f z = 
-- [(x,y) | x <- [u .. z] , y <- [v .. 0] , f (x , y) == z]

invert f z = find (0 , z) f z

find (u , v) f z 
     | u > z && v < 0 = []
     | z' < z         = find (u + 1 , v) f z
     | z' == z        = (u , v) : find (u + 1 , v - 1) f z
     | z' > z         = find (u , v - 1) f z 
     where  z' = f (u , v)

-- 以左上为起始点, 只有两个方向, 一方通行遍历可能性. 
-- 向前(u+1), 向下(v-1), 对应于严格单调递增函数 $f$ 值的增减调节.
-- ver. Theo

-- m = max $ filter ((\y -> f(0,y) <= z) [0 .. z]
-- n = max $ filter ((\x -> f(x,0) <= z) [0 .. z]
-- replace the (z+1)*(z+1) square with an (n+1)*(m+1) square

invert f z = find (0 , m) f z

find (u , v) f z 
     | u > n && v < 0 = []
     | z' < z         = find (u + 1 , v) f z
     | z' == z        = (u , v) : find (u + 1 , v - 1) f z
     | z' > z         = find (u , v - 1) f z 
     where  z' = f (u , v)

-- 改进了x, y的上界. 左上角(0,m), 右下角(n,0).

-- compute m and n by binary search
-- m = bsearch g (x , y) z
-- determine m s.t. x <= m < y, g m <= z < g (m + 1)

bsearch g (a , b) z
        | a + 1 == b  = a
        | g m <= z    = bsearch g (m , b) z
        | otherwise   = bsearch g (a , m) z
        where m = (a + b) `div` 2

m = bsearch (\y -> f (0 , y)) (- 1 , z + 1) z
n = bsearch (\x -> f (x , 0)) (- 1 , z + 1) z

-- (- 1 , z + 1)
--  x <= m < y
--  z + 1 so m be able to take the upper bound z
--  -1 so the first m = (- 1 + (z + 1)) / 2 = (0 + z) / 2

if $f (p , q) < z$, throw away lower-left $A$
if $f (p , q) > z$, throw away upper-right $B$
if $f (p , q) = z$, throw away $A$ and $B$

左上: $(u,v)$, 右下: $(r,s)$
行数: $r-u$, 列数: $s-v$

若列数大于行数, 从中间行 $q$ 开始进行 rfind, bsearch q $\rightarrow $ p, 二分查找列 $p$, s.t. $f(p,q) \leq z < f(p+1,q)$;
若$f(p,q) < z$, 在矩形区域 $((p+1,q-1),(r,s))$ 和 $((u,v) , (p,q+1))$ 中继续查找;
若$f(p,q) = z$, 则取出列 $p$ , 在矩形区域 $((p+1,q-1) , (r,s))$ 和 $((u,v) , (p-1,q+1))$ 中继续查找;
若行数大于列数, 从中间列 $p$ 开始进行 cfind, bsearch p $\rightarrow $ q, 二分查找行 $q$ , s.t. $f(p,q) \leq z < f(p,q+1)$;
若$f(p,q) < z$, 在矩形区域 $((u,v),(p-1,q+1))$ 和 $((p+1,q) , (r,s))$ 中继续查找;
若$f(p,q) = z$, 取出行 $q$ ,在矩形区域 $((u , v) , (p - 1, q + 1))$ 和 $((p + 1 , q - 1) , (r , s))$ 中继续查找.

(tips: 行号对应纵坐标, 列号对应横坐标, 和矩阵序号方法相反;
行和列的加减与函数值的加减一致, (单变量递增函数性质), 由此易写出子查找矩形区域.)

-- ver. Marry

find (u , v) (r , s) f z
     | u > r && v < s = []
     | v - s <= r - u = rfind $ bsearch (\x -> f (x , q)) (u - 1 , r + 1) $ z
     | otherwise      = cfind $ bsearch (\y -> f (p , y)) (s - 1 , v + 1) $ z
     where 
           p = (u + v) `div` 2
           q = (v + s) `div` 2
           rfind p = find (p + 1 , q - 1) (r , s) f z ++
                     if f (p , q) == z 
                     then (p , q) : find (u , v) (p - 1 , q + 1) f z
                     else find (u , v) (p , q + 1) f z 
           cfind q = find (u , v) (p - 1, q + 1) f z ++ 
                     if f (p , q) == z 
                     then (p , q) : find (p + 1 , q - 1) (r , s) f z
                     else find (p + 1 , q) (r , s) f z

invert f z = find (0 , m) (n , 0) f z
             where m = bsearch (\y -> f(0 , y)) (-1 , z + 1) z
                   n = bsearch (\x -> f(x , 0)) (-1 , z + 1) z
